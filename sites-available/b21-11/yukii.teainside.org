
server {
	listen 80;
	listen [::]:80;
	server_name yukii.teainside.org;
	root /var/app/yukii/public;
	error_log /var/app/yukii/logs/error.log;
	access_log /var/app/yukii/logs/access.log;

	rewrite ^/(.*)$ https://yukii.teainside.org/$1 permanent;
}

server {
	ssl on;
	listen 443;
	listen [::]:443;

	server_name yukii.teainside.org;
	root /var/app/yukii/public;
	error_log /var/app/yukii/logs/error.log;
	access_log /var/app/yukii/logs/access.log;

	ssl_certificate /var/app/ssl/veno.site/crtca;
	ssl_certificate_key /var/app/ssl/veno.site/key;

	index index.php;

	location / {
		try_files $uri $uri/ =404;
	}

	location ~ \.php$ {
		include snippets/fastcgi-php.conf;
		fastcgi_pass unix:/run/php/php7.2-fpm.sock;
	}
}
