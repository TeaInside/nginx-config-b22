
server {
	listen 80;
	listen [::]:80;
	server_name webhook.teainside.org;

	rewrite ^/(.*)$ https://webhook.teainside.org/$1 permanent;
}

server {
	listen 443;
	listen [::]:443;

	server_name webhook.teainside.org;

	ssl on;
	ssl_certificate /var/www/.ssl/@@.teainside.tech/crtca;
	ssl_certificate_key /var/www/.ssl/@@.teainside.tech/key;

	root /var/www/webhook.teainside.org/public;
	error_log /var/www/webhook.teainside.org/logs/error.log;
	access_log /var/www/webhook.teainside.org/logs/access.log;

	index index.php;

	location / {
		autoindex off;
		try_files $uri $uri/ =404;
	}

	location ~ \.php$ {
		include snippets/fastcgi-php.conf;
		fastcgi_pass unix:/var/run/php/php7.2-fpm.sock;
	}
}
